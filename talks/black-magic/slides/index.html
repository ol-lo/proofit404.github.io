def get_caller():
    f = sys._getframe()
    while hasattr(f, 'f_code'):
        filename = f.f_code.co_filename
        if filename == __file__:
            f = f.f_back
            continue
        return f.f_code.co_name

class StaticMethodDescriptor:
    def __init__(self, f):
        self.f = f
    def __get__(self, obj, cls=None):
        f = self.__dict__['f']
        return partial(f, cls)
