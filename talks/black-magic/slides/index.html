def get_caller():

        if filename == __file__:
            f = f.f_back
            continue
        return f.f_code.co_name

ast rewrite

import ast

class Int2Str(ast.NodeTransformer):

    def visit_Num(self, node):
        return ast.copy_location(
            ast.Str(s=str(node.n)),
            node,
        )

how it works

exp = 'c = [1, 2, 3]'
exec(exp)
print(c)
# [1, 2, 3]
ast = ast.parse(exp)
new_exp = Int2Str().visit(ast)
exec(compile(new_exp))
print(c)
# ['1', '2', '3']
