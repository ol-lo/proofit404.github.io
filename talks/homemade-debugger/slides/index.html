Frame evaluation

_PyEval_EvalFrameDefault(PyFrameObject *f, int throwflag) {
     switch (opcode) {
          TARGET(CALL_FUNCTION) {
               PyObject **sp, *res;
               res = call_function(&sp, oparg, NULL);
               stack_pointer = sp;
               PUSH(res);
          }
     }
}
